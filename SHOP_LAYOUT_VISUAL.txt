KEY DESIGN PRINCIPLES:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✓ ASYMMETRY: No two rows look the same - creates visual interest
✓ RHYTHM: Pattern of 7-5-4-5-3-6 columns repeats per category
✓ VARIETY: Mix of landscape, square, and portrait aspect ratios
✓ BREATHING ROOM: 32px (mt-32) between category sections
✓ HIERARCHY: Category titles (text-5xl) > Product titles (text-2xl)
✓ ELEGANCE: Cormorant Garamond serif + neutral palette
✓ HOVER EFFECTS: Subtle scale-105 zoom on images
✓ RESPONSIVE: Full-width on mobile, asymmetric on desktop

MOBILE BEHAVIOR:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

On mobile (< 768px), all items stack vertically at full width:
